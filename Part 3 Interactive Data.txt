/* ======================================================
   SCROLL REVEAL — watches for sections entering viewport
====================================================== */
const revealObserver = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      entry.target.classList.add('visible');
      // Trigger bar chart animation when Cost Wall section is visible
      if (entry.target.id === 'tuition-chart' || entry.target.closest('#cost-wall')) {
        animateTuitionBars();
      }
      // Trigger cost card bars
      if (entry.target.classList.contains('cost-card')) {
        const fill = entry.target.querySelector('.cost-bar-fill');
        if (fill) fill.style.width = fill.dataset.w;
      }
    }
  });
}, { threshold: 0.15 });

document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));

    }
  });
}, { threshold: 0.4 });

/* ======================================================
   SECTION 4: DONUT CHART HELPER
====================================================== */
function drawDonut(pct) {
  // pct = { personal, us, other } — should sum to 100
  const svg = document.getElementById('donut-svg');
  const legend = document.getElementById('donut-legend');
  const cx = 100, cy = 100, r = 75, strokeW = 34;
  const circumference = 2 * Math.PI * r;

  const slices = [
    { key: 'personal', label: 'Personal / Family',         color: 'var(--orange)', val: pct.personal },
    { key: 'us',       label: 'U.S. Funding (Scholarship)',color: 'var(--blue)',   val: pct.us       },
    { key: 'other',    label: 'Other Sources',             color: 'var(--muted)', val: pct.other    },
  ];

  // Build SVG
  let svgHTML = `<circle cx="${cx}" cy="${cy}" r="${r}"
    fill="none" stroke="rgba(255,255,255,0.07)" stroke-width="${strokeW}" />`;

  let offset = -90; // start at top

  slices.forEach(slice => {
    const ratio = slice.val / 100;
    const dash  = ratio * circumference;
    const gap   = circumference - dash;
    const rotate = offset;
    svgHTML += `<circle cx="${cx}" cy="${cy}" r="${r}"
      fill="none"
      stroke="${slice.color}"
      stroke-width="${strokeW}"
      stroke-dasharray="${dash} ${gap}"
      stroke-dashoffset="${-rotate / 360 * circumference}"
      stroke-linecap="butt"
      style="transform-origin:center;transform:rotate(${rotate}deg);transition:stroke-dasharray 0.7s ease"
    />`;
    offset += ratio * 360;
  });

  // Center label
  svgHTML += `<text x="${cx}" y="${cy - 6}" text-anchor="middle"
    font-family="Playfair Display,serif" font-size="22" font-weight="700"
    fill="white">${pct.personal}%</text>
  <text x="${cx}" y="${cy + 16}" text-anchor="middle"
    font-family="DM Sans,sans-serif" font-size="9.5" fill="rgba(200,230,247,0.7)">SELF-FUNDED</text>`;

  svg.innerHTML = svgHTML;

  // Build legend
  legend.innerHTML = slices.map(s => `
    <div class="legend-item">
      <div class="legend-dot" style="background:${s.color}"></div>
      <span>${s.label}</span>
      <span class="legend-pct">${s.val}%</span>
    </div>
  `).join('');
}
